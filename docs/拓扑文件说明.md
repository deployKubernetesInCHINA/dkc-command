拓扑文件编辑说明
--

文件是 inventory/hosts.yaml, 当前拓扑文件是8个服务器, 其中有3 master, 3 node, 3 mongo (arbiter复用surpass-master-1).
可修改项说明
- 服务器的登录用户名/密码/IP/Port
- kube-master组可以是1个或3个节点, 建议是3个保证高可用
- sg1组, 定义为业务服务组
- rs1组, 是mongo集群. 服务器只属于rs1组,则认为mongo是独占,cache设置为当前服务器内存的60%,否则设置为30%

```yaml
all:
  vars:
    external_domain_name: "www.test.com"     #ingress 使用的域名
  hosts:
    localhost:
      ansible_connection: local
      remote_machine_username: "root"        #远程节点的登录用户名
      remote_machine_password: "123456"      #远程节点的登录密码
    surpass-master-1:                        #远程节点列表，ip、端口都可以改，
      ansible_host: 192.168.57.3
      ansible_port: 22
    surpass-master-2:
      ansible_host: 192.168.57.4
      ansible_port: 22
    surpass-master-3:
      ansible_host: 192.168.57.5
      ansible_port: 22
    surpass-worker-1:
      ansible_host: 192.168.57.6
      ansible_port: 22
    surpass-worker-2:
      ansible_host: 192.168.57.7
      ansible_port: 22
    surpass-worker-3:
      ansible_host: 192.168.57.8
      ansible_port: 22
    surpass-worker-4:
      ansible_host: 192.168.57.9
      ansible_port: 22
    surpass-worker-5:
      ansible_host: 192.168.57.10
      ansible_port: 22
  children:
    kube-master:                               #k8s cluster中3个master节点列表
      hosts:                                   #此列表hostname可以更改，但是要保持数量是3
        surpass-master-1:
        surpass-master-2:
        surpass-master-3:
    om:                                        #mornitoring 节点列表，此列表会部署k8s node组件，并且设置nodeLabel nodeType: "om"
      hosts:
        surpass-master-3:
    sg1:                                       #service group1 节点列表，此列表会部署k8s node组件
      hosts:                                   #此列表可以更改
        surpass-worker-1:
        surpass-worker-2:
        surpass-worker-3:
    sg:                                        #service group 节点列表组，无需改动
      children:
        sg1:
    rs1:                                       #mongodb 集群节点列表，此列表会部署k8s node组件，并且设置taints："db.surpass=true:NoExecute"
      vars:
        rs1_port: 30003                        #mongodb rs0 端口
      hosts:                                   #节点列表中，hostname可以改，PSA顺序请不要改动。
        surpass-worker-4:                      #mongodb primary节点
          rs1_role: "primary"
        surpass-worker-5:                      #mongodb secondary节点
          rs1_role: "secondary"
        surpass-master-1:                      #mongodb arbiter节点
          rs1_role: "arbiter"
    rs2:
      vars:
        rs2_port: 30004
      hosts:
        surpass-worker-1:
          rs2_role: "primary"
    db:
      children:
        rs1:
        rs2:
    kube-node:                                 #k8s node节点列表组，无需改动
      children:
        om:
        sg:
        db:
    k8s-cluster:                               #k8s 集群列表组， 无需改动
      children:
        kube-master:
        kube-node:

```
